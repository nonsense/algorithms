### Problem 250

Solved for 75, after many mistakes.

{% highlight cpp %}
class WakingUp
{
  private:
    vector<int> period;
    vector<int> start;
    vector<int> volume;

    int D;

  public:
    bool wouldHeWakeUpFor(int S)
    {
      int time = 0;
      int sleepiness = S;

      while (sleepiness > 0)
      {
        time++;
        sleepiness += D;

        for (int i=0; i<period.size(); i++)
        {
          if (start[i] == time || ((time-start[i]) % period[i]) == 0)
            sleepiness -= volume[i];
        }

        // LCM of 1, 2, 3, 4, 5, 6, 7, 8, 9
        if (time == 2520 && S > sleepiness)
          return true;
        else if (time == 2520 && S <= sleepiness)
          return false;
      }

      return true;
    }

    int maxSleepiness(vector<int> p, vector<int> s, vector<int> v, int d)
    {
      period = p;
      start = s;
      volume = v;
      D = d;

      int hi = numeric_limits<int>::max();
      int lo = 0;

      bool wontWakeUpForAll = false;

      int best = 0;

      while (lo <= hi)
      {
        int pivot = lo + (hi-lo)/2;

        if (wouldHeWakeUpFor(pivot))
        {
          best = pivot;

          lo = pivot+1;
        }
        else
        {
          wontWakeUpForAll = true;

          hi = pivot-1;
        }
      }

      if (!wontWakeUpForAll)
        return -1;

      return best;
    }
};
